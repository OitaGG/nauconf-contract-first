Пример спецификации

      
        openapi: 3.0.2
        info:
          description: Описание списка эндпоинтов
          title: Заголовок
          version: Версия
        paths:
          /someEntity/{entityUuid}:
            parameters:
                - $ref: "#/components/parameters/entityUuidParams"
            put:
              summary: Отредактировать сущность
              operationId: editInstallation
              requestBody:
                description: Информация для обновления сущности
                required: true
                content:
                  application/json:
                    schema:
                      $ref: "#/components/schemas/SomeEntityDto"
              responses:
                200:
                  description: Сущность обновлена
                  content:
                    application/json:
                      schema:
                        $ref: "#/components/schemas/SomeEntityDto"
                404:
                  description: Сущность не найдена
                500:
                  description: Ошибка при редактировании сущности
